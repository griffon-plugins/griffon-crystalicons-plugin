16 package griffon.plugins.crystalicons;
17 
18 import javax.annotation.Nonnull;
19 
20 import static griffon.util.GriffonClassUtils.requireState;
21 import static griffon.util.GriffonNameUtils.isBlank;
22 
23 /**
24  * @author Andres Almiray
25  */
26 public final class Crystal {
27     private final String category;
28     private final String description;
29 
30     public Crystal(@Nonnull String category, @Nonnull String description) {
31         this.category = category;
32         this.description = description;
33     }
34 
35     @Nonnull
36     public String getCategory() {
37         return category;
38     }
39 
40     @Nonnull
41     public String getDescription() {
42         return description;
43     }
44 
45     @Nonnull
46     public String formatted() {
47         return category + ":" + description;
48     }
49 
50     @Nonnull
51     public String asResource(int size) {
52         requireState(size == 16 || size == 24 || size == 32, "Argument 'size' must be one of [16, 22, 32].");
53         return "com/everaldo/crystal/" + size + "x" + size + "/" + category + "/" + description + ".png";
54     }
55 
56     public boolean equals(Object o) {
57         if (o == null) return false;
58         if (o == this) return true;
59         if (!(o instanceof Crystal)) return false;
60         Crystal c = (Crystal) o;
61         return category.equals(c.category) && description.equals(c.description);
62     }
63 
64     public int hashCode() {
65         return (category.hashCode() * 37) + (description.hashCode() * 31);
66     }
67 
68     @Nonnull
69     public static Crystal findByDescription(@Nonnull String description) {
70         if (isBlank(description)) {
71             throw new IllegalArgumentException("Description " + description + " is not a valid Crystal icon description");
72         }
73         if (!description.contains(":")) {
74             throw new IllegalArgumentException("Description " + description + " is not a valid Crystal icon description");
75         }
76 
77         ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
78 
79         String[] parts = description.split(":");
80         Crystal icon = new Crystal(parts[0].toLowerCase(), parts[1].toLowerCase());
81         if (classLoader.getResource(icon.asResource(16)) != null) {
82             return icon;
83         } else if (classLoader.getResource(icon.asResource(24)) != null) {
84             return icon;
85         } else if (classLoader.getResource(icon.asResource(32)) != null) {
86             return icon;
87         }
88 
89         throw new IllegalArgumentException("Description " + description + " is not a a valid Crystal icon description");
90     }
91 }
