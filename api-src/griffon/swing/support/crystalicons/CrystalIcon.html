16 package griffon.swing.support.crystalicons;
17 
18 import griffon.plugins.crystalicons.Crystal;
19 
20 import javax.annotation.Nonnull;
21 import javax.swing.ImageIcon;
22 import java.net.URL;
23 
24 import static java.util.Objects.requireNonNull;
25 
26 /**
27  * @author Andres Almiray
28  */
29 public class CrystalIcon extends ImageIcon {
30     private final Crystal crystal;
31     private final int size;
32 
33     public CrystalIcon(@Nonnull Crystal crystal) {
34         this(crystal, 16);
35     }
36 
37     public CrystalIcon(@Nonnull Crystal crystal, int size) {
38         super(toURL(crystal, size));
39         this.crystal = crystal;
40         this.size = size;
41     }
42 
43     public CrystalIcon(@Nonnull String description) {
44         this(Crystal.findByDescription(description));
45     }
46 
47     @Nonnull
48     private static URL toURL(@Nonnull Crystal crystal, int size) {
49         requireNonNull(crystal, "Argument 'crystal' must not be null.");
50         String resource = crystal.asResource(size);
51         URL url = Thread.currentThread().getContextClassLoader().getResource(resource);
52         if (url == null) {
53             throw new IllegalArgumentException("Icon " + crystal.formatted() + " does not exist");
54         }
55         return url;
56     }
57 
58     @Nonnull
59     public Crystal getCrystal() {
60         return crystal;
61     }
62 
63     public int getSize() {
64         return size;
65     }
66 }
